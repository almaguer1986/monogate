\section{Fully Verified Theorems (0 sorries)}

\subsection{Lower Bounds — SuperBEST F16 Ground-Truth}

\begin{description}
\item[SB(neg) $\geq$ 2 (T09)]
  \lean{NegLowerBound.lean}{SB\_neg\_ge\_two}
  No single F16 operator computes $-x$ for all real $x$. Proof: exhaustive witness for
  each of 16 operators. Theorem: $\neg\,\exists\,\text{op}\in F16,\;\forall x,\;\text{op}(x) = -x$.

\item[SB(add) $\geq$ 2 (ADD-T1)]
  \lean{AddLowerBound.lean}{SB\_add\_ge\_two}
  No single F16 operator computes $x+y$ for all real $x,y$.

\item[SB(sub) $\geq$ 2 (T33)]
  \lean{SubLowerBound.lean}{SB\_sub\_ge\_two}
  No single F16 operator computes $x-y$ for all real $x,y$.

\item[SB(mul, general) $\geq$ 2 (T\_MUL\_GEN\_LB)]
  \lean{MulLowerBound.lean}{SB\_mul\_ge\_two}
  No single F16 operator computes $x \cdot y$ for all real $x,y$.
  (Positive domain: 1 node via $F_{16}(x,y) = \exp(\ln x + \ln y) = xy$, proved separately.)

\item[SB(div, general) $\geq$ 2 (T\_DIV\_GEN\_LB)]
  \lean{DivLowerBound.lean}{SB\_div\_ge\_two}
  No single F16 operator computes $x/y$ for all real $x,y$.
\end{description}

\subsection{Upper Bounds — Positive Domain 1-Node Constructions}

\begin{description}
\item[SB(exp) = 1 (T\_EXP\_1N)]
  \lean{UpperBounds.lean}{exp\_one\_node}
  $\exp(x) - \ln(1) = \exp(x)$ for all $x \in \mathbb{R}$. Single F16 node F1.

\item[SB(mul, $x,y>0$) = 1 (T\_MUL\_1N)]
  \lean{UpperBounds.lean}{mul\_one\_node\_positive}
  $\exp(\ln x + \ln y) = xy$ for $x,y > 0$. Single F16 node F16fn.
  Also proved in \lean{ModelAudit.lean}{mul\_is\_one\_node\_positive}.

\item[SB(pow, $x>0$) = 1 (T\_POW\_1N)]
  \lean{UpperBounds.lean}{rpow\_one\_node\_positive}
  $\exp(n \ln x) = x^n$ for $x > 0$, all $n \in \mathbb{R}$. Single EPL node.

\item[SB(recip, $x>0$) = 1 (T\_RECIP\_1N)]
  \lean{UpperBounds.lean}{recip\_one\_node\_positive}
  $\exp(-\ln x) = 1/x$ for $x > 0$. EPL with $n = -1$.

\item[SB(sqrt, $x>0$) = 1 (T\_SQRT\_1N)]
  \lean{UpperBounds.lean}{sqrt\_one\_node\_positive'}
  Also \lean{ModelAudit.lean}{sqrt\_is\_one\_node\_positive}.
  $\exp(\tfrac{1}{2}\ln x) = \sqrt{x}$ for $x > 0$. EPL with $n = 1/2$.
\end{description}

\subsection{Euler / Depth Hierarchy Theorems}

\begin{description}
\item[Euler Gateway (T03, T04)]
  \lean{EMLDepth.lean}{euler\_gateway}
  $\text{ceml}(ix, 1) = \exp(ix)$ for all $x \in \mathbb{R}$.
  Corollary: $\exp(x) \in \text{EML-1}$ (depth-1 EML tree; definitional from eval).

\item[Euler Identity (T05)]
  \lean{EMLDepth.lean}{euler\_identity}
  $\text{ceml}(i\pi, 1) + 1 = 0$, i.e., $e^{i\pi} + 1 = 0$. Follows from
  $\text{Complex.exp\_pi\_mul\_I}$ in Mathlib.

\item[Strict Depth Hierarchy: EML-0 $\subsetneq$ EML-1 (T06)]
  \lean{EMLDepth.lean}{exp\_not\_constant}
  The function $\exp(x)$ is EML-1 but is not constant, witnessing $\text{EML-0} \subsetneq \text{EML-1}$.

\item[sin has infinitely many zeros (T01a)]
  \lean{InfiniteZerosBarrier.lean}{sin\_has\_infinitely\_many\_zeros}
  The set $\{x \in \mathbb{R} \mid \sin(x) = 0\}$ is infinite. Explicit construction:
  $\sin(n\pi) = 0$ for all $n \in \mathbb{Z}$ (\texttt{sin\_int\_pi\_zero}, also verified).
  This is Part A of the Infinite Zeros Barrier; Parts B and C remain sorry'd.
\end{description}
