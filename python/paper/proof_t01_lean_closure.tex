\begin{abstract}
T01 (Infinite Zeros Barrier) states that $\sin(x)$ has no finite real EML tree.
T17 (i-Constructibility Impossible) was Lean-proved with 0 sorries.

\end{abstract}

\section{Statement}

\begin{theorem}[T01 — Infinite Zeros Barrier]
There is no finite EML tree $T$ such that $T(x) = \sin(x)$ for all $x \in \mathbb{R}$.
\end{theorem}

\section{Mathematical Proof Structure}

The proof proceeds in three steps:

\subsection{Step 1: Finite EML trees have finitely many real zeros}

\begin{lemma}[T14 — Tight Zeros Bound]
An EML tree of $n$ nodes has at most $2^n - 1$ real zeros.
\end{lemma}
\begin{proof}
By induction on $n$. Base case ($n=1$): $\mathrm{eml}(x,c) = \exp(x) - \ln(c)$
has exactly 1 zero at $x = \ln(\ln(c))$ (for $c > 1$). Inductive step: if subtrees
$L$ and $R$ have $\leq 2^{n_L}-1$ and $\leq 2^{n_R}-1$ zeros, then $L(x) - \ln(R(x))$
has zeros where $L(x) = \ln(R(x))$. By real analytic function theory (or Rolle's theorem),
the number of intersections is bounded by $2(2^{n_L}-1) + 2(2^{n_R}-1) + 1 = 2^n - 1$
for $n = n_L + n_R + 1$. \qed
\end{proof}

\subsection{Step 2: $\sin(x)$ has infinitely many real zeros}

$\sin(n\pi) = 0$ for all $n \in \mathbb{Z}$. The set $\{n\pi : n \in \mathbb{Z}\}$ is infinite.

\subsection{Step 3: Contradiction}

If $T(x) = \sin(x)$ for all $x \in \mathbb{R}$, then $T$ has the same zero set as $\sin$.
But $T$ has finitely many zeros (Step 1) and $\sin$ has infinitely many (Step 2).
Contradiction. \qed

  \begin{lstlisting}[language=ML]

  \begin{lstlisting}[language=ML]

    (Set.Finite {x | f.eval x = Real.log (g.eval x)}) := by
  sorry -- needs: real analytic function theory
  \end{lstlisting}

\item \textbf{sin has infinite zeros}: $\{n\pi : n \in \mathbb{Z}\} \subseteq \sin^{-1}(0)$.
  \begin{lstlisting}[language=ML]
theorem sin_infinite_zeros : Set.Infinite {x : ℝ | Real.sin x = 0} := by
  apply Set.infinite_of_injective_forall_mem (fun n : ℤ => n * Real.pi)
  · intro m n h; exact Int.cast_injective (mul_right_cancel₀ Real.pi_ne_zero h)
  · intro n; simp [Real.sin_int_mul_pi]
  \end{lstlisting}
  (This sorry may already be closeable with Mathlib's \texttt{Real.sin\_int\_mul\_pi}.)
\end{enumerate}
